\begin{center}
    \large{\textbf{Applied Linear Algebra in Data Analysis}}
\end{center}
\hrule
\vspace{0.5cm}

\noindent \textbf{Vector spaces}: A set of vectors that is closed under vector scaling and vector addition. E.g. $\R^n$, $\C^n$. \textit{A vector space will always contain the zero vector.}

\noindent \textbf{Subspace}: A subset of a vector space $\mc{V}$ which is also a vector space.

\noindent \textbf{Span of a set} (of vectors): The set of all linear combinations of a set of vectors $\mc{S} = \lc \mf{s}_i \rc_{i=1}^{p}$ from the vector space $\mc{V}$.
\[ \text{span } \mc{S} = \lc \sum_{i=1}^p \alpha_i \mf{s}_i \,\, \Big \vert \,\, \alpha_i \in \R \rc \subseteq \mc{V}\]

\noindent \textbf{Linear Independence}: A set $\mc{S}$ is linearly independent if and only if, $\sum_{i=1}^p \alpha_i \mf{s}_i = \mf{0} \implies \alpha_i = 0, \forall i$. \textit{If the set has $\mf{0}$, then the set is linearly dependent.}

\noindent \textbf{Basis}: A set of vectors $\mc{B}$ is a basis for a vector space $\mc{V}$ if and only if, $\mc{B}$ is linearly independent and $\text{span }\mc{B} = \mc{V}$. The elements of $\mc{B}$ are called basis vectors of $\mc{V}$. \textit{There are infinitely many bases for a vector space. Every vector in $\mc{V}$ can be written as a \textbf{unique} linear combination of the basis vectors.}

\noindent \textbf{Dimension}: The number of basis vectors in a basis of a vector space $\mc{V}$ is called the dimension of $\mc{V}$.

\noindent \textbf{Inner product}: $\mf{x}^\top\mf{y} = \sum_{i=1}^n x_i y_i, \,\, \mf{x}, \mf{y} \in \R^n$.

\noindent \textbf{Norm}: Measure of the length of a vector. $\lV \mf{x} \rV_p = \lp \sum_{i=1}^n \vert x_i \vert^p \rp^{\frac{1}{p}}, \,\, \mf{x} \in \R^n$. $\Vert \mf{x} \Vert_2^2 = \mf{x}^\top \mf{x}$.

\noindent \textbf{Cauchy-Schwarz Inequality}: $\vert \mf{x}^\top \mf{y} \vert \leq \lV \mf{x} \rV_2 \lV \mf{y} \rV_2$.

\noindent \textbf{Orthogonality}: Two vectors $\mf{x}, \mf{y} \in \R^n$ are orthogonal if and only if, $\mf{x}^\top \mf{y} = 0$.

\noindent \textbf{Orthonormal basis}: A basis $\mc{B} = \lc \mf{b}_i \rc_{i=1}^n$ is orthonormal if and only if, $\mf{b}_i^\top \mf{b}_j = \delta_{ij}$, where $\delta_{ij}$ is the Kronecker delta function.

\noindent \textbf{Linear Function}: A function $f: \R^n \to \R$ that satisfies superposition. All linear functions $f$ can be represented as $f(\mf{x}) = \mf{w}^\top \mf{x}$, where $\mf{w} \in \R^n$.
